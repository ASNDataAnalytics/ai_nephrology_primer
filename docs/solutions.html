#| eval: false
/*=============================================================
  CLINICAL TRIAL ELIGIBILITY: Black/AA Adults with CKD 1–4
  Sedentary Lifestyle Intervention Study
  
  Inclusion:
    - Self-identified Black/African American
    - Age 35–80
    - CKD Stage 1–4 (eGFR ≥15, no ESRD/transplant)
    - Has a telephone on file
  Exclusion:
    - Kidney transplant (Hx or active)
    - Lower extremity amputation (w/o prosthetic) — flagged via Dx
    - Orthostatic hypotension Dx
    - Unstable angina/CAD Dx
    - Severe cognitive impairment Dx
    - Active psychiatric/neurologic preclusion Dx
    - Wheelchair/walker/assistive device dependency (ADL note flags)
    - Participated in Patient Advisory Board (research flag)
  
  Notes on sedentary time and functional mobility:
    These are self-reported and typically NOT structured in Clarity.
    Flag them for manual/survey screening as noted below.
=============================================================*/

WITH

-- ── 1. DEMOGRAPHICS: Black/AA, Age 35–80 ─────────────────────────────────────
eligible_demographics AS (
    SELECT
        p.PAT_ID,
        p.PAT_MRN_ID,
        p.PAT_FIRST_NAME,
        p.PAT_LAST_NAME,
        p.BIRTH_DATE,
        DATEDIFF(YEAR, p.BIRTH_DATE, GETDATE())
            - CASE WHEN DATEADD(YEAR, DATEDIFF(YEAR, p.BIRTH_DATE, GETDATE()), p.BIRTH_DATE) > GETDATE()
                   THEN 1 ELSE 0 END                        AS AGE_YEARS,
        p.SEX_C,
        zr.NAME                                             AS RACE_NAME,
        p.PAT_STATUS                                        -- 1 = Alive
    FROM PATIENT p
    JOIN ZC_PATIENT_RACE zpr  ON zpr.PAT_ID = p.PAT_ID
    JOIN ZC_RACE          zr  ON zr.RACE_C  = zpr.RACE_C
    WHERE
        zr.NAME IN (
            'Black or African American',
            'Black/African American',
            'African American'          -- varies by site mapping
        )
        AND DATEDIFF(YEAR, p.BIRTH_DATE, GETDATE())
            - CASE WHEN DATEADD(YEAR, DATEDIFF(YEAR, p.BIRTH_DATE, GETDATE()), p.BIRTH_DATE) > GETDATE()
                   THEN 1 ELSE 0 END  BETWEEN 35 AND 80
        AND p.PAT_STATUS = 1           -- living patients only
),

-- ── 2. TELEPHONE ON FILE ──────────────────────────────────────────────────────
has_phone AS (
    SELECT DISTINCT PAT_ID
    FROM PAT_PHONE
    WHERE PHONE_TYPE_C IN (1, 2, 3)    -- Home=1, Work=2, Mobile=3; verify your site's ZC_PHONE_TYPE
      AND PHONE_NUM IS NOT NULL
      AND PHONE_NUM <> ''
),

-- ── 3. CKD STAGE 1–4 ─────────────────────────────────────────────────────────
-- Primary approach: most recent eGFR lab value
--   CKD Stage 1: eGFR ≥90 (requires albuminuria/structural marker — see note)
--   CKD Stage 2: eGFR 60–89
--   CKD Stage 3a: 45–59  | Stage 3b: 30–44
--   CKD Stage 4: 15–29
--   CKD Stage 5/ESRD: <15 → EXCLUDE
--
-- Secondary approach: active problem list CKD ICD-10 codes (see codebook below)
--
-- NOTE: Stage 1 without a documented CKD Dx code is hard to distinguish from
--       normal renal function purely from eGFR. Consider requiring a CKD Dx.

ckd_egfr AS (
    SELECT
        r.PAT_ID,
        r.RESULT_TIME,
        TRY_CAST(r.ORD_VALUE AS FLOAT)  AS EGFR_VALUE,
        ROW_NUMBER() OVER (
            PARTITION BY r.PAT_ID
            ORDER BY r.RESULT_TIME DESC
        )                               AS RN
    FROM ORDER_RESULTS r
    JOIN CLARITY_COMPONENT cc
        ON cc.COMPONENT_ID = r.COMPONENT_ID
    WHERE
        -- Common LOINC codes for eGFR (CKD-EPI, MDRD variants)
        cc.LNC_CODE IN (
            '33914-3',   -- eGFR MDRD
            '50044-7',   -- eGFR CKD-EPI (Females)
            '48642-3',   -- eGFR non-Black MDRD
            '48643-1',   -- eGFR Black MDRD
            '62238-1',   -- eGFR CKD-EPI 2009
            '98979-8'    -- eGFR CKD-EPI 2021 (race-free)
        )
        AND r.RESULT_STATUS_C = 3       -- Final result
        AND TRY_CAST(r.ORD_VALUE AS FLOAT) IS NOT NULL
),

ckd_stage AS (
    SELECT
        PAT_ID,
        EGFR_VALUE,
        CASE
            WHEN EGFR_VALUE >= 90 THEN 'Stage 1 (eGFR only — confirm Dx)'
            WHEN EGFR_VALUE BETWEEN 60 AND 89.9 THEN 'Stage 2'
            WHEN EGFR_VALUE BETWEEN 45 AND 59.9 THEN 'Stage 3a'
            WHEN EGFR_VALUE BETWEEN 30 AND 44.9 THEN 'Stage 3b'
            WHEN EGFR_VALUE BETWEEN 15 AND 29.9 THEN 'Stage 4'
            WHEN EGFR_VALUE < 15              THEN 'Stage 5/ESRD — EXCLUDE'
        END                             AS CKD_STAGE_LABEL,
        RESULT_TIME                     AS EGFR_DATE
    FROM ckd_egfr
    WHERE RN = 1
),

-- CKD confirmed via active problem list (belt-and-suspenders)
ckd_dx AS (
    SELECT DISTINCT pl.PAT_ID
    FROM PROBLEM_LIST pl
    JOIN CL_ICD10_PX icd ON icd.CODE = pl.ICD10_CODE
    WHERE
        pl.PROBLEM_STATUS_C = 1             -- Active
        AND pl.ICD10_CODE IN (
            -- CKD Stage 1–4 ICD-10 codes
            'N18.1','N18.2','N18.30','N18.31','N18.32','N18.4',
            -- Diabetic CKD Stage 1–4
            'E11.21','E11.22','E10.21','E10.22',
            -- Hypertensive CKD Stage 1–4
            'I12.9','I13.10'
        )
),

-- ── 4. EXCLUSION: Kidney Transplant ──────────────────────────────────────────
transplant_excl AS (
    SELECT DISTINCT pl.PAT_ID
    FROM PROBLEM_LIST pl
    WHERE pl.PROBLEM_STATUS_C = 1
      AND pl.ICD10_CODE IN (
          'Z94.0',    -- Kidney transplant status
          'T86.10','T86.11','T86.12','T86.13','T86.19'  -- Transplant complications
      )
    UNION
    -- Also check surgical history
    SELECT DISTINCT sh.PAT_ID
    FROM HSP_ACCT_PX_LIST sh
    WHERE sh.PX_ID IN (
        SELECT PX_ID FROM CL_PX
        WHERE PX_CODE IN ('50360','50365','50370','50380')  -- CPT: kidney transplant
    )
),

-- ── 5. EXCLUSION: Lower-Extremity Amputation ─────────────────────────────────
amputation_excl AS (
    SELECT DISTINCT PAT_ID
    FROM PROBLEM_LIST
    WHERE PROBLEM_STATUS_C = 1
      AND ICD10_CODE IN (
          -- AKA (above-knee)
          'Z89.611','Z89.612','Z89.619',
          -- BKA (below-knee)
          'Z89.511','Z89.512','Z89.519'
      )
    -- NOTE: Prosthetic use is NOT structured in Clarity; manual chart review needed
),

-- ── 6. EXCLUSION: Orthostatic Hypotension ────────────────────────────────────
ortho_htn_excl AS (
    SELECT DISTINCT PAT_ID
    FROM PROBLEM_LIST
    WHERE PROBLEM_STATUS_C = 1
      AND ICD10_CODE IN ('I95.1', 'I95.0', 'I95.8', 'I95.9')
),

-- ── 7. EXCLUSION: Unstable CAD / Angina ──────────────────────────────────────
unstable_cad_excl AS (
    SELECT DISTINCT PAT_ID
    FROM PROBLEM_LIST
    WHERE PROBLEM_STATUS_C = 1
      AND ICD10_CODE IN (
          'I20.0',    -- Unstable angina
          'I24.0',    -- Acute coronary thrombosis (no MI)
          'I24.8','I24.9'  -- Other/unspecified ACS
      )
),

-- ── 8. EXCLUSION: Severe Cognitive Impairment ────────────────────────────────
cognitive_excl AS (
    SELECT DISTINCT PAT_ID
    FROM PROBLEM_LIST
    WHERE PROBLEM_STATUS_C = 1
      AND ICD10_CODE IN (
          'F01.51','F01.50',   -- Vascular dementia
          'F02.80','F02.81',   -- Dementia in other diseases
          'F03.90','F03.91',   -- Unspecified dementia
          'G30.0','G30.1','G30.8','G30.9',  -- Alzheimer's
          'G31.09',            -- Frontotemporal dementia
          'F10.27','F10.97'    -- Alcohol-related dementia
      )
),

-- ── 9. EXCLUSION: Relevant Neurologic/Psychiatric Conditions ─────────────────
neuro_psych_excl AS (
    SELECT DISTINCT PAT_ID
    FROM PROBLEM_LIST
    WHERE PROBLEM_STATUS_C = 1
      AND ICD10_CODE IN (
          -- Parkinson's
          'G20','G21.11','G21.19',
          -- ALS
          'G12.21',
          -- Severe schizophrenia/psychosis flags
          'F20.0','F20.1','F20.2','F20.3','F20.5','F20.9',
          -- Active mania/psychotic bipolar
          'F31.2','F31.5'
      )
),

-- ── 10. EXCLUSION: Orthopedic Preclusion ─────────────────────────────────────
orthopedic_excl AS (
    SELECT DISTINCT PAT_ID
    FROM PROBLEM_LIST
    WHERE PROBLEM_STATUS_C = 1
      AND ICD10_CODE IN (
          'M96.0',    -- Pseudarthrosis post-fusion (spinal)
          'M43.00',   -- Spondylolysis
          'M48.06',   -- Spinal stenosis lumbar
          'M47.816',  -- Spondylosis w/ radiculopathy, lumbar
          -- Severe hip/knee OA with functional limitation
          'M16.0','M16.10','M16.11','M16.12',  -- Primary OA hip
          'M17.0','M17.10','M17.11','M17.12'   -- Primary OA knee
          -- ⚠ Expand per clinical judgment; not exhaustive
      )
),

-- ── FINAL ASSEMBLY ─────────────────────────────────────────────────────────--
final_cohort AS (
    SELECT
        d.PAT_ID,
        d.PAT_MRN_ID,
        d.PAT_FIRST_NAME,
        d.PAT_LAST_NAME,
        d.BIRTH_DATE,
        d.AGE_YEARS,
        d.RACE_NAME,
        cs.EGFR_VALUE,
        cs.CKD_STAGE_LABEL,
        cs.EGFR_DATE,
        -- Flag for manual screening items (not structured in Clarity)
        1 AS NEEDS_SURVEY_SEDENTARY_TIME,   -- Self-report: ≥6 hrs/day sedentary
        1 AS NEEDS_SURVEY_MOBILITY,          -- Self-report: able to walk 1 block, stand unassisted
        1 AS NEEDS_SURVEY_ASSISTIVE_DEVICE,  -- Self-report: no wheelchair/walker/cane dependency
        1 AS NEEDS_CHART_REVIEW_PAB          -- Confirm not in Patient Advisory Board
    FROM eligible_demographics d
    INNER JOIN has_phone            hp  ON hp.PAT_ID = d.PAT_ID
    -- Require CKD eGFR 15–89 OR confirmed CKD Dx on problem list
    INNER JOIN ckd_stage            cs  ON cs.PAT_ID = d.PAT_ID
                                       AND cs.EGFR_VALUE BETWEEN 15 AND 89.9
    -- CKD Dx belt-and-suspenders (optional: use INNER JOIN to require both)
    -- INNER JOIN ckd_dx            cd  ON cd.PAT_ID = d.PAT_ID
    -- Exclusions
    LEFT  JOIN transplant_excl      tx  ON tx.PAT_ID  = d.PAT_ID
    LEFT  JOIN amputation_excl      ax  ON ax.PAT_ID  = d.PAT_ID
    LEFT  JOIN ortho_htn_excl       ox  ON ox.PAT_ID  = d.PAT_ID
    LEFT  JOIN unstable_cad_excl    cx  ON cx.PAT_ID  = d.PAT_ID
    LEFT  JOIN cognitive_excl       cog ON cog.PAT_ID = d.PAT_ID
    LEFT  JOIN neuro_psych_excl     np  ON np.PAT_ID  = d.PAT_ID
    LEFT  JOIN orthopedic_excl      oe  ON oe.PAT_ID  = d.PAT_ID
    WHERE
        tx.PAT_ID  IS NULL   -- no transplant
        AND ax.PAT_ID  IS NULL   -- no AKA/BKA
        AND ox.PAT_ID  IS NULL   -- no orthostatic hypotension
        AND cx.PAT_ID  IS NULL   -- no unstable CAD
        AND cog.PAT_ID IS NULL   -- no severe cognitive impairment
        AND np.PAT_ID  IS NULL   -- no preclusive neuro/psych Dx
        AND oe.PAT_ID  IS NULL   -- no preclusive orthopedic Dx
)

SELECT *
FROM final_cohort
ORDER BY AGE_YEARS, PAT_MRN_ID;
Column Name Description
PATIENT.PAT_ID VARCHAR — Epic internal unique patient identifier (primary key across most patient-centric tables).
PATIENT.PAT_MRN_ID VARCHAR — Medical Record Number; the human-facing patient identifier used in clinical workflows.
PATIENT.PAT_FIRST_NAME VARCHAR — Patient’s legal first name.
PATIENT.PAT_LAST_NAME VARCHAR — Patient’s legal last name.
PATIENT.BIRTH_DATE DATE — Patient date of birth; used to derive AGE_YEARS.
PATIENT.PAT_STATUS INT (category) — Vital status. 1 = Living; 2 = Deceased. Sourced from ZC_PAT_STATUS.
ZC_PATIENT_RACE.PAT_ID VARCHAR — Foreign key linking patient to their race record(s); supports multi-race entries (one row per race).
ZC_RACE.RACE_C INT (category) — Numeric race category code. Joins to ZC_PATIENT_RACE and ZC_RACE.
ZC_RACE.NAME VARCHAR — Display name of the race category (e.g., 'Black or African American'). Site-specific string values; validate against your local ZC_RACE table.
PAT_PHONE.PHONE_TYPE_C INT (category) — Phone type code. Typical values: 1 = Home, 2 = Work, 3 = Mobile. Confirm mapping in ZC_PHONE_TYPE.
PAT_PHONE.PHONE_NUM VARCHAR — Stored phone number string; used to confirm telephone access.
ORDER_RESULTS.COMPONENT_ID INT — Links lab result row to the specific analyte (e.g., eGFR). Joins to CLARITY_COMPONENT.
ORDER_RESULTS.ORD_VALUE VARCHAR — Raw result value as entered/reported (must be cast to FLOAT for numeric comparison).
ORDER_RESULTS.RESULT_TIME DATETIME — Timestamp when the lab result was resulted/verified. Used to identify the most recent eGFR.
ORDER_RESULTS.RESULT_STATUS_C INT (category) — Result status. 3 = Final; 1 = Preliminary. Always filter to Final (3) for clinical use.
CLARITY_COMPONENT.LNC_CODE VARCHAR — LOINC code for the lab component. Used to identify eGFR-specific assays (e.g., '62238-1' = CKD-EPI eGFR).
PROBLEM_LIST.PAT_ID VARCHAR — Foreign key to PATIENT; links diagnoses to a patient.
PROBLEM_LIST.ICD10_CODE VARCHAR — ICD-10-CM diagnosis code on the problem list (e.g., 'N18.3' = CKD Stage 3).
PROBLEM_LIST.PROBLEM_STATUS_C INT (category) — Problem list entry status. 1 = Active; 2 = Resolved; 3 = Deleted. Filter to 1 for active conditions.
HSP_ACCT_PX_LIST.PX_ID INT — Foreign key to CL_PX; identifies a procedure performed during a hospital account. Used to detect transplant CPT codes.
CL_PX.PX_CODE VARCHAR — CPT or ICD procedure code. Used to identify kidney transplant procedures (e.g., '50360').
EGFR_VALUE (derived) FLOAT — Most recent eGFR result cast from ORD_VALUE. CKD stages mapped: Stage 1 ≥90, Stage 2 = 60–89, Stage 3a = 45–59, Stage 3b = 30–44, Stage 4 = 15–29, Stage 5 <15.
CKD_STAGE_LABEL (derived) VARCHAR — Human-readable CKD stage label derived from eGFR value. For screening classification only.
AGE_YEARS (derived) INT — Calculated age in years as of query run date using BIRTH_DATE.
NEEDS_SURVEY_* (derived flags) BIT (1) — Placeholder flags indicating criteria that cannot be ascertained from Clarity and require patient self-report (sedentary time, functional mobility, assistive device use) or manual chart/registry review (PAB participation).

⚠️ Important Caveats

  1. Sedentary time (≥6 hrs/day) is self-reported and has no structured Clarity field. It must be collected via screening survey or REDCap instrument.

  2. Functional mobility (ability to walk a block, stand unassisted, assistive device use) is similarly self-reported and inconsistently coded in Clarity. Social history free text (SOCIAL_HX) or flowsheet rows may have partial data, but manual screening is the reliable approach.

  3. Patient Advisory Board participation is not a standard Clarity field. Check your institution’s research registry (OnCore, REDCap, or a custom research table) for this exclusion.

  4. Race coding is site-dependent. Run SELECT DISTINCT NAME FROM ZC_RACE to confirm the exact string(s) used at your institution before deploying.

  5. eGFR Stage 1 (≥90) without an explicit CKD diagnosis on the problem list is indistinguishable from normal renal function by eGFR alone. Consider requiring a corroborating N18.1 ICD-10 code or albuminuria documentation.

Here’s a complete script using Polars and scikit-learn to predict remission status:

Code 
import polars as pl
import numpy as np
from sklearn.ensemble import RandomForestClassifier, GradientBoostingClassifier
from sklearn.linear_model import LogisticRegression
from sklearn.model_selection import StratifiedKFold, cross_validate
from sklearn.pipeline import Pipeline
from sklearn.impute import SimpleImputer
from sklearn.preprocessing import StandardScaler
from sklearn.metrics import (
    classification_report, roc_auc_score, average_precision_score,
    make_scorer, f1_score
)
import warnings
warnings.filterwarnings("ignore")

# ── 1. Load data ──────────────────────────────────────────────────────────────
df = pl.read_csv("your_data.csv")          # ← adjust path / format as needed

# ── 2. Define features & target ───────────────────────────────────────────────
TARGET = "remission"

FEATURES = [
    "days_of_life", "plt", "mpv", "un", "wbc", "hgb", "hct", "rbc",
    "mcv", "mch", "mchc", "rdw",
    "neut_percent", "lymph_percent", "mono_percent", "eos_percent", "baso_percent",
    "sod", "pot", "chlor", "co2", "creat", "gluc", "cal",
    "prot", "alb", "ast", "alt", "alk", "tbil",
    "active",        # clinical flag — strong prior candidate
]

# ── 3. Polars preprocessing ───────────────────────────────────────────────────
# Log-transform right-skewed lab values to reduce outlier influence
skewed = ["ast", "alt", "alk", "tbil", "gluc", "wbc", "plt", "un", "creat"]

df = df.with_columns([
    pl.col(c).log1p().alias(c) for c in skewed if c in df.columns
])

# Extract numpy arrays (Polars → numpy for sklearn)
X = df.select(FEATURES).to_numpy().astype(np.float64)
y = df[TARGET].to_numpy().astype(int)

print(f"Dataset shape : {X.shape}")
print(f"Class balance : {np.bincount(y)} (0=no remission, 1=remission)")

# ── 4. Build pipelines ────────────────────────────────────────────────────────
# Median imputation handles the missing values described above; scaler for LR
imputer  = SimpleImputer(strategy="median")
scaler   = StandardScaler()

pipelines = {
    "Logistic Regression": Pipeline([
        ("impute", imputer),
        ("scale",  scaler),
        ("clf",    LogisticRegression(max_iter=1000, class_weight="balanced",
                                      C=0.1, solver="lbfgs")),
    ]),
    "Random Forest": Pipeline([
        ("impute", imputer),
        ("clf",    RandomForestClassifier(
            n_estimators=500, max_features="sqrt",
            class_weight="balanced", random_state=42, n_jobs=-1
        )),
    ]),
    "Gradient Boosting": Pipeline([
        ("impute", imputer),
        ("clf",    GradientBoostingClassifier(
            n_estimators=300, learning_rate=0.05,
            max_depth=3, subsample=0.8, random_state=42
        )),
    ]),
}

# ── 5. Cross-validation ───────────────────────────────────────────────────────
cv = StratifiedKFold(n_splits=5, shuffle=True, random_state=42)

scorers = {
    "roc_auc"  : "roc_auc",
    "avg_prec" : "average_precision",
    "f1"       : make_scorer(f1_score, zero_division=0),
    "accuracy" : "accuracy",
}

results = {}
for name, pipe in pipelines.items():
    cv_res = cross_validate(pipe, X, y, cv=cv, scoring=scorers,
                            return_train_score=False)
    results[name] = {k: cv_res[f"test_{k}"] for k in scorers}
    print(f"\n{'─'*50}")
    print(f"  {name}")
    print(f"{'─'*50}")
    for metric, vals in results[name].items():
        print(f"  {metric:<12}  {vals.mean():.3f}  ±  {vals.std():.3f}")

# ── 6. Feature importance (Random Forest) ────────────────────────────────────

Key design decisions explained:

Missing data — SimpleImputer(strategy="median") handles all the missingness you described. Median is preferred over mean here because several lab values (AST, ALT, ALK, BUN) are heavily right-skewed.

Log transforms — Applied via Polars before the sklearn pipeline for AST, ALT, ALK, tbil, glucose, WBC, platelets, BUN, and creatinine. These are notorious for extreme outliers in clinical labs that can destabilize linear models and impair tree splits.

Class imbalance — class_weight="balanced" is set for both Logistic Regression and Random Forest, which automatically up-weights the minority class. GBM handles this implicitly through its loss function but you can add scale_pos_weight if imbalance is severe.

Three model types — Logistic Regression gives interpretable coefficients and a calibration baseline; Random Forest is robust to multicollinearity (which will be present among the CBC components like HGB/HCT/RBC); Gradient Boosting often wins on tabular clinical data.

Evaluation metrics — ROC-AUC, Average Precision (PR-AUC), and F1 are all reported because accuracy alone is misleading with class imbalance.

active feature — Included intentionally since active inflammation is clinically the most direct predictor of remission failure; if you want a “lab-only” model, simply remove it from FEATURES.

    if ismissing(x)

# Parse DOB from multiple common Excel/string formats
function parse_dob(x)
    if ismissing(x)
        return missing
    end
    # Already a Date/DateTime
    if x isa Date
        return x
    elseif x isa DateTime
        return Date(x)
    end

    s = strip(String(x))
    if isempty(s)
        return missing
    end

    # Normalize month-name strings like "may 5 1970" -> "May 5 1970"
    s2 = join(uppercasefirst.(split(lowercase(s), ' ')), ' ')

    # Try a few formats
    for fmt in (dateformat"y-m-d", dateformat"m/d/y", dateformat"u d y", dateformat"u d, y")
        try
            return Date(s2, fmt)
        catch
        end
    end

    # Last resort: try Date(s2) (can work for some ISO-ish strings)
    try
        return Date(s2)
    catch
        return missing
    end
end

function iqr_bounds(v::Vector{Union{Missing, Float64}})
    x = collect(skipmissing(v))
    if length(x) < 4
        return (missing, missing)  # too few points
    end
    q1 = quantile(x, 0.25)
    q3 = quantile(x, 0.75)
    iqr = q3 - q1
    return (q1 - 1.5*iqr, q3 + 1.5*iqr)
end

function df_to_md_table(d::DataFrame; max_rows::Int=50)
    n = min(nrow(d), max_rows)
    cols = names(d)
    header = "|" * join(string.(cols), "|") * "|\n"
    sep    = "|" * join(fill("---", length(cols)), "|") * "|\n"
    rows = ""
    for i in 1:n
        vals = [ismissing(d[i, c]) ? "missing" : string(d[i, c]) for c in cols]
        rows *= "|" * join(vals, "|") * "|\n"
    end
    if nrow(d) > max_rows
        rows *= "\n*(truncated to first $(max_rows) rows)*\n"
    end
    return header * sep * rows
end
